// Ограничение частоты запросов
function checkRateLimit($action, $limit = 5, $timeWindow = 300)
{
    if (!isset($_SESSION['rate_limit'][$action])) {
        $_SESSION['rate_limit'][$action] = [];
    }

    $now = time();
    $_SESSION['rate_limit'][$action] = array_filter(
        $_SESSION['rate_limit'][$action],
        function ($timestamp) use ($now, $timeWindow) {
            return $timestamp > $now - $timeWindow;
        }
    );

    if (count($_SESSION['rate_limit'][$action]) >= $limit) {
        return false;
    }

    $_SESSION['rate_limit'][$action][] = $now;
    return true;
}

// Проверка номера телефона
function validatePhone($phone)
{
    $digitsOnly = preg_replace('/[^0-9]/', '', $phone);

    if (strlen($digitsOnly) < 10 || strlen($digitsOnly) > 12) {
        return false;
    }

    $firstDigit = substr($digitsOnly, 0, 1);
    if (!in_array($firstDigit, ['7', '8', '9'])) {
        return false;
    }

    return true;
}

// Форматирование номера телефона
function formatPhone($phone)
{
    $digitsOnly = preg_replace('/[^0-9]/', '', $phone);

    if (strlen($digitsOnly) == 10) {
        return '+7' . $digitsOnly;
    } elseif (strlen($digitsOnly) == 11 && $digitsOnly[0] == '7') {
        return '+' . $digitsOnly;
    } elseif (strlen($digitsOnly) == 11 && $digitsOnly[0] == '8') {
        return '+7' . substr($digitsOnly, 1);
    } else {
        return '+' . $digitsOnly;
    }
}

/**
 * Генерирует пазл: фон с дырой + фрагмент (изображение уменьшается до 500px по ширине)
 * @return array с base64-данными изображений и координатами отверстия
 */
function generatePuzzleCaptcha()
{
    $uploadsDir = __DIR__ . '/uploads/albums/';
    $sourceImages = glob($uploadsDir . '*.{jpg,jpeg,png,gif,webp}', GLOB_BRACE);

    // Загружаем исходное изображение
    if (empty($sourceImages)) {
        $img = imagecreatetruecolor(400, 300);
        imagefill($img, 0, 0, imagecolorallocate($img, 200, 200, 200));
    } else {
        $srcPath = $sourceImages[array_rand($sourceImages)];
        $imageData = file_get_contents($srcPath);
        if ($imageData === false) {
            $img = imagecreatetruecolor(400, 300);
            imagefill($img, 0, 0, imagecolorallocate($img, 200, 200, 200));
        } else {
            $img = imagecreatefromstring($imageData);
            if (!$img) {
                $img = imagecreatetruecolor(400, 300);
                imagefill($img, 0, 0, imagecolorallocate($img, 200, 200, 200));
            }
        }
    }

    // Масштабируем изображение до разумного размера (максимум 500px по ширине)
    $srcWidth = imagesx($img);
    $srcHeight = imagesy($img);
    $maxWidth = 500;
    if ($srcWidth > $maxWidth) {
        $newWidth = $maxWidth;
        $newHeight = intval($srcHeight * ($maxWidth / $srcWidth));
        $resized = imagecreatetruecolor($newWidth, $newHeight);
        imagecopyresampled($resized, $img, 0, 0, 0, 0, $newWidth, $newHeight, $srcWidth, $srcHeight);
        imagedestroy($img);
        $img = $resized;
        $width = $newWidth;
        $height = $newHeight;
    } else {
        $width = $srcWidth;
        $height = $srcHeight;
    }

    // Размер вырезаемого куска (сделаем пропорциональным ширине, например 20% от ширины)
    $pieceW = intval($width * 0.2);
    $pieceH = $pieceW; // квадрат

    // Случайная позиция выреза (отступы от краёв)
    $holeX = rand(30, $width - $pieceW - 30);
    $holeY = rand(30, $height - $pieceH - 30);

    // Фон с дырой
    $bg = imagecreatetruecolor($width, $height);
    imagecopy($bg, $img, 0, 0, 0, 0, $width, $height);
    $holeColor = imagecolorallocate($bg, 220, 220, 220);
    imagefilledrectangle($bg, $holeX, $holeY, $holeX + $pieceW, $holeY + $pieceH, $holeColor);

    // Фрагмент
    $piece = imagecreatetruecolor($pieceW, $pieceH);
    imagecopy($piece, $img, 0, 0, $holeX, $holeY, $pieceW, $pieceH);

    // Кодируем в base64
    ob_start();
    imagepng($bg);
    $bgData = ob_get_clean();
    ob_start();
    imagepng($piece);
    $pieceData = ob_get_clean();

    // Сохраняем в сессии координаты отверстия
    $_SESSION['puzzle_hole'] = ['x' => $holeX, 'y' => $holeY, 'w' => $pieceW, 'h' => $pieceH];
    $_SESSION['puzzle_solved'] = false;

    return [
        'bg' => 'data:image/png;base64,' . base64_encode($bgData),
        'piece' => 'data:image/png;base64,' . base64_encode($pieceData),
        'holeX' => $holeX,
        'holeY' => $holeY,
        'pieceW' => $pieceW,
        'pieceH' => $pieceH,
    ];
}

/**
 * Проверяет, что пользователь правильно собрал пазл
 * @return bool
 */
function verifyPuzzleCaptcha()
{
    if (!isset($_SESSION['puzzle_hole']) || !isset($_SESSION['puzzle_solved'])) {
        return false;
    }
    $solved = $_SESSION['puzzle_solved'];
    unset($_SESSION['puzzle_hole']);
    unset($_SESSION['puzzle_solved']);
    return $solved;
}

// Ограничение частоты запросов
function checkRateLimit($action, $limit = 5, $timeWindow = 300)
{
    if (!isset($_SESSION['rate_limit'][$action])) {
        $_SESSION['rate_limit'][$action] = [];
    }

    $now = time();
    $_SESSION['rate_limit'][$action] = array_filter(
        $_SESSION['rate_limit'][$action],
        function ($timestamp) use ($now, $timeWindow) {
            return $timestamp > $now - $timeWindow;
        }
    );

    if (count($_SESSION['rate_limit'][$action]) >= $limit) {
        return false;
    }

    $_SESSION['rate_limit'][$action][] = $now;
    return true;
}

/**
 * Генерирует пазл: фон с дырой + фрагмент (изображение уменьшается до 500px по ширине)
 * @return array с base64-данными изображений и координатами отверстия
 */
function generatePuzzleCaptcha()
{
    $uploadsDir = __DIR__ . '/uploads/albums/';
    $sourceImages = glob($uploadsDir . '*.{jpg,jpeg,png,gif,webp}', GLOB_BRACE);

    // Загружаем исходное изображение
    if (empty($sourceImages)) {
        $img = imagecreatetruecolor(400, 300);
        imagefill($img, 0, 0, imagecolorallocate($img, 200, 200, 200));
    } else {
        $srcPath = $sourceImages[array_rand($sourceImages)];
        $imageData = file_get_contents($srcPath);
        if ($imageData === false) {
            $img = imagecreatetruecolor(400, 300);
            imagefill($img, 0, 0, imagecolorallocate($img, 200, 200, 200));
        } else {
            $img = imagecreatefromstring($imageData);
            if (!$img) {
                $img = imagecreatetruecolor(400, 300);
                imagefill($img, 0, 0, imagecolorallocate($img, 200, 200, 200));
            }
        }
    }

    // Масштабируем изображение до разумного размера (максимум 500px по ширине)
    $srcWidth = imagesx($img);
    $srcHeight = imagesy($img);
    $maxWidth = 500;
    if ($srcWidth > $maxWidth) {
        $newWidth = $maxWidth;
        $newHeight = intval($srcHeight * ($maxWidth / $srcWidth));
        $resized = imagecreatetruecolor($newWidth, $newHeight);
        imagecopyresampled($resized, $img, 0, 0, 0, 0, $newWidth, $newHeight, $srcWidth, $srcHeight);
        imagedestroy($img);
        $img = $resized;
        $width = $newWidth;
        $height = $newHeight;
    } else {
        $width = $srcWidth;
        $height = $srcHeight;
    }

    // Размер вырезаемого куска (сделаем пропорциональным ширине, например 20% от ширины)
    $pieceW = intval($width * 0.2);
    $pieceH = $pieceW; // квадрат

    // Случайная позиция выреза (отступы от краёв)
    $holeX = rand(30, $width - $pieceW - 30);
    $holeY = rand(30, $height - $pieceH - 30);

    // Фон с дырой
    $bg = imagecreatetruecolor($width, $height);
    imagecopy($bg, $img, 0, 0, 0, 0, $width, $height);
    $holeColor = imagecolorallocate($bg, 220, 220, 220);
    imagefilledrectangle($bg, $holeX, $holeY, $holeX + $pieceW, $holeY + $pieceH, $holeColor);

    // Фрагмент
    $piece = imagecreatetruecolor($pieceW, $pieceH);
    imagecopy($piece, $img, 0, 0, $holeX, $holeY, $pieceW, $pieceH);

    // Кодируем в base64
    ob_start();
    imagepng($bg);
    $bgData = ob_get_clean();
    ob_start();
    imagepng($piece);
    $pieceData = ob_get_clean();

    // Сохраняем в сессии координаты отверстия
    $_SESSION['puzzle_hole'] = ['x' => $holeX, 'y' => $holeY, 'w' => $pieceW, 'h' => $pieceH];
    $_SESSION['puzzle_solved'] = false;

    return [
        'bg' => 'data:image/png;base64,' . base64_encode($bgData),
        'piece' => 'data:image/png;base64,' . base64_encode($pieceData),
        'holeX' => $holeX,
        'holeY' => $holeY,
        'pieceW' => $pieceW,
        'pieceH' => $pieceH,
    ];
}

/**
 * Проверяет, что пользователь правильно собрал пазл
 * @return bool
 */
function verifyPuzzleCaptcha()
{
    if (!isset($_SESSION['puzzle_hole']) || !isset($_SESSION['puzzle_solved'])) {
        return false;
    }
    $solved = $_SESSION['puzzle_solved'];
    unset($_SESSION['puzzle_hole']);
    unset($_SESSION['puzzle_solved']);
    return $solved;
}

    // Валидация длины
    if (strlen($nickname) < 2 || strlen($nickname) > 50) {
        $_SESSION['error'] = 'Имя должно содержать от 2 до 50 символов.';
        redirect('index.php');
    }

    if (strlen($content) < 5 || strlen($content) > 1000) {
        $_SESSION['error'] = 'Отзыв должен содержать от 5 до 1000 символов.';
        redirect('index.php');
    }

    // Запрещаем HTML теги и ссылки
    if ($nickname !== strip_tags($nickname) || $content !== strip_tags($content)) {
        $_SESSION['error'] = 'HTML теги и ссылки запрещены.';
        redirect('index.php');
    }

    // Проверка на спам (повторяющиеся символы)
    if (preg_match('/(.)\1{4,}/', $content)) {
        $_SESSION['error'] = 'Слишком много повторяющихся символов.';

    }

    // Проверка на ссылки
    if (preg_match('/https?:\/\/|www\./i', $content)) {

        redirect('index.php');
    }

    // Вставляем отзыв
    try {
        $stmt = $pdo->prepare("INSERT INTO review (nickname, content) VALUES (?, ?)");
        $stmt->execute([$nickname, $content]);

        $_SESSION['success'] = 'Спасибо! Отзыв отправлен на модерацию.';
    } catch (PDOException $e) {
        error_log('Review insert error: ' . $e->getMessage());

    }

} else {
    redirect('index.php');

session_start();

header('Content-Type: application/json');

if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['action']) && $_POST['action'] === 'solve') {
    // Проверяем CSRF-токен, если хотите повысить безопасность
    if (!isset($_POST['csrf_token']) || !verifyCSRFToken($_POST['csrf_token'])) {
        echo json_encode(['success' => false, 'error' => 'CSRF token mismatch']);
        exit;
    }
    $_SESSION['puzzle_solved'] = true;
    echo json_encode(['success' => true]);
} else {
    echo json_encode(['success' => false]);
